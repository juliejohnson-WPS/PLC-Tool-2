<script>

    const totalSteps = 9;


    function updateProgressBar(currentStep) {

        const percentage = ((currentStep - 1) / (totalSteps - 1)) * 100;

        document.getElementById('progressBar').style.width = percentage + '%';

    }


    function nextStep(stepNumber) {

        // Validation check for the currently active step inputs

        const currentActive = document.querySelector('.step-panel.active');

        const inputs = currentActive.querySelectorAll('input, textarea');

        let allValid = true;


        inputs.forEach(input => {

            if (!input.checkValidity()) {

                input.reportValidity();

                allValid = false;

            }

        });


        if (!allValid) return;


        // Transition views

        currentActive.classList.remove('active');

        document.getElementById('step' + stepNumber).classList.add('active');

        updateProgressBar(stepNumber);

    }


    function prevStep(stepNumber) {

        document.querySelector('.step-panel.active').classList.remove('active');

        document.getElementById('step' + stepNumber).classList.add('active');

        updateProgressBar(stepNumber);

    }


    function generatePlan() {

        // Double-check validation for step 8 before continuing

        const currentActive = document.querySelector('.step-panel.active');

        const textarea = currentActive.querySelector('textarea');

        if (!textarea.checkValidity()) {

            textarea.reportValidity();

            return;

        }


        // Gather Values

        const grade = document.getElementById('gradeLevel').value;

        const course = document.getElementById('course').value;

        const eq = document.getElementById('essentialQuestion').value;

        const knowDo = document.getElementById('q1KnowDo').value;

        const becoming = document.getElementById('q1Becoming').value;

        const measure = document.getElementById('q2Measure').value;

        const intervention = document.getElementById('q3NotProficient').value;

        const extension = document.getElementById('q4Proficient').value;

        const reflect = document.getElementById('plcReflection').value;


        // Map Values to Output Section

        document.getElementById('outContext').innerText = `${grade} — ${course}`;

        document.getElementById('outFocus').innerText = eq;

        document.getElementById('outPlanKnow').innerText = knowDo;

        document.getElementById('outPlanBecome').innerText = becoming;

        document.getElementById('outAssess').innerText = measure;

        document.getElementById('outIntervention').innerText = intervention;

        document.getElementById('outExtension').innerText = extension;

        document.getElementById('outReflect').innerText = reflect;


        // Advance to final panel

        nextStep(9);

    }
